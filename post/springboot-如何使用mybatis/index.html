1.反向生成#

使用mybatis-plugins插件可以实现自动生成

注意:自动生成使用驼峰原则,下划线后的字母变为大写,其他大写字母变为小写.

添加依赖:插件

            <plugin>
                <groupId>org.mybatis.generator</groupId>
                <artifactId>mybatis-generator-maven-plugin</artifactId>
                <version>1.3.2</version>
                <configuration>
                    <verbose>true</verbose>
                    <overwrite>true</overwrite>
                </configuration>
            </plugin>

配置generatorConfig.xml

  • 只看和修改带序号的即可
  • 多次使用插件,xml里面的代码会叠加,记住删除或查看
  • 在table下面

2.mapper.xml#

1.xml原本结构#

<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE mapper PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN" "http://mybatis.org/dtd/mybatis-3-mapper.dtd">
<!--namespace : mapper接口层-->
<!--resultType : 实体类层-->
<mapper namespace="com.bootdo.system.dao.TableDescriptionDao">
    <select id="list" resultType="com.bootdo.system.domain.TableDescriptionDO">
        SELECT * from table where id =#{id}
   </select>
</mapper>

3.条件查询语句#

主要使用一个全新的接口:selectBySelective,最重要的是xml的动态语句

<select id="selectBySelective" parameterType="com.ericsson.model.kpi.实体对象" resultType="com.ericsson.model.kpi.实体对象">
    select  *
    from udc_kpi_formulaName
    <where>
        <!--注意 Integer类型 和 String类型的不同之处-->
      <if test="id != null" >   
        id=#{id,jdbcType=INTEGER}
      </if>
      <if test="name != null and name !=''" >
        and name = #{name,jdbcType=VARCHAR}
      </if>
      <if test="true">
        and deleteFlag =1
      </if>
    </where>
  </select>

注意:

  int类型: <if test="INTEGER != null"> 
String类型: <if test="STRING != null and STRING !='' ">

4.分页#

service#
    @Override
    public PageBean findAll(实体对象 record, Page page) {
        com.github.pagehelper.Page<Object> pageHelper = PageHelper.startPage(page.getPageIndex(), page.getPageSize());
        List<实体对象> list = 实体Mapper.selectBySelective(record);
        //返回pageBean对象
        int pageNum = pageHelper.getPageNum();
        long total = pageHelper.getTotal();
        PageBean pageBean = new PageBean();

        if (total % pageHelper.getPageSize() == 0 && total / pageHelper.getPageSize() + 1 == pageNum) {
            com.github.pagehelper.Page<Object> pageHelper1 = PageHelper.startPage(page.getPageIndex() - 1, page.getPageSize());
            List<实体对象> list1 = 实体Mapper.selectBySelective(record);
            pageBean.setPage(String.valueOf(pageHelper1.getPageNum()));
            pageBean.setRows(list1);
            pageBean.setTotal(total);
        } else {
            pageBean.setPage(String.valueOf(pageNum));
            pageBean.setRows(list);
            pageBean.setTotal(total);
        }
        return pageBean;
    }

5.批量插入#

mapper接口#
 int insertBatch(List<entity> entitys);
 int updateBatch(List<entity> entitys);
mapper.xml#
  <insert id="insertBatch" parameterType="java.util.List">
    insert into 表格 (id,name,相关字段
    )
    values
    <foreach collection="list" item="entitys(传来的list参数名)" index="index" separator=",">
      (#{entitys.id,jdbcType=INTEGER},
       #{entitys.name,jdbcType=VARCHAR},
       #{entitys.相关字段,jdbcType=VARCHAR}
      )
    </foreach>
  </insert>
  <update id="updateBatch" parameterType="java.util.List">
    update 表格
    set 列名=
    <foreach collection="list" item="entitys(传来的list参数名)" index="index"
             separator=" " open="case ID" close="end">
      when #{entitys.id} then #{entitys.列名}
    </foreach>
    where id in
    <foreach collection="list" index="index" item="entitys"
             separator="," open="(" close=")">
      #{entitys.id,jdbcType=BIGINT}
    </foreach>
  </update>

关于批处理查询的问题: 批处理当超过一定的次数就无法执行

 //将List 分成多个List   
private static List<List<Object>> splitList(List<Object> messagesList, int groupSize) {
        int length = messagesList.size();
        // 计算可以分成多少组
        int num = (length + groupSize - 1) / groupSize; 
        List<List<Object>> newList = new ArrayList<>(num);
        for (int i = 0; i < num; i++) {
            // 开始位置
            int fromIndex = i * groupSize;
            // 结束位置
            int toIndex = (i + 1) * groupSize < length ? (i + 1) * groupSize : length;
            newList.add(messagesList.subList(fromIndex, toIndex));
        }
        return newList;
    }

6.查询是否重名#

主要要注意的是新增时的查询和修改时的查询是不一样的,修改时查询记得把自己这一行去除掉

<select id="selectNumByName" parameterType="com.ericsson.model.kpi.实体对象" resultType="Integer">
    select count(*) from  表
    <where>
      <if test="name !=null and name!=''">
        name=#{name,jdbcType=VARCHAR}
      </if>
      <if test="id != null">
        and id !=#{id,jdbcType=INTEGER}
      </if>
    </where>
  </select>

7.如何获得刚刚插入的键值的id#

mapper接口

    @Insert("insert into 表格 (name,其他字段) values(#{neName},#{其他字段})")
    @Options(useGeneratedKeys = true,keyProperty = "id(参数值不是table值)",keyColumn = "id(数据库字段值)")
    int insertReturnPrimaryKey(实体对象 record);

或者可以使用xml

<!-- useGeneratedKeys="true" keyProperty="id"(此处时参数值,不是table值)  -->
<insert id="save" useGeneratedKeys="true" keyProperty="id" parameterType="com.bootdo.business.domain.TableDO">
</insert>

service实现

Mapper.insertReturnPrimaryKey(record);
 //record(插入的主体)
int id=record.getId();  //获得刚插入的主键
